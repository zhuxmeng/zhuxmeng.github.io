框架

c++：

using namespace std;

int main(){

system("pause");

return 0;

}

c:
#include<stdio.h>
int main(){

return 0;

}

注释:
单行注释 //
多行注释 /* */

输入与输出

cout<<” “<<endl; printf(“”);
cout <<”a = “<<a<<endl; printf(“a = %d”,a)
cin>>a; scanf(“%d”,&a);
cin>>str; scanf(“%s”,&str)

常量

宏常量
#define day 7
const修饰常量
const int MONTH = 12; //一般用全大写强调常量

变量

数据类型 变量名 = 变量初始值
int a =10;

数据类型

整型
int num1 = 10;

size of关键字 可以计算数据类型所占内存空间
cout<<””<<sizeof(num1)<<endl;
printf(“%d”,sizeof num1);

浮点型
单精度 float 6位小数
双精度 double 15位小数
float f1 =3.14f； float f1 = 3.14；

字符型
char ch = ‘a’; 单个字符

转义字符
换行 \n
反斜杠 \
水平制表符 \t

字符串型
char a[] = “abc”; string a = “abc”;(需要使用#include头文件)

布尔类型
bool flag = true;

运算符

算数运算符
+ - * /
取模 %
递增 ++
递减 –

程序流结构

switch语句
switch(表达式)
{
case 结果1;执行语句; break;
case 结果2;执行语句; break;
…
default;执行语句; break;
}

常见函数

遍历字符串
string a = “abc”;
for(int c=0;c < a.size();c++)
{
cout<<s[i];
}

c_str()
int main() {
std::string str = “Hello, World!”;

// 使用 c_str() 方法获取 C 风格字符串
const char* charArray = str.c_str();

// 输出字符数组
std::cout << "charArray: " << charArray << std::endl;

return 0;

}

pow
int main() {
double base = 2.0;
double exponent = 3.0;

// 使用 pow 函数计算次方
double result = std::pow(base, exponent);

std::cout << base << " raised to the power of " << exponent << " is " << result << std::endl;

return 0;

}

stoi
int main() {
std::string str = “123”;
int num = std::stoi(str);
std::cout << “转换后的整数: “ << num << std::endl;
return 0;
}

指针

const修饰指针

常量指针 const int * p = &a; //指针指向的值不可改，指针的指向可改
指针常量 int * const p = &a; //指针指向的值可以改，指针的指向不可以改

指针可以通过形参修饰实参
